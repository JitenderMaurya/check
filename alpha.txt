def determine_rotation(words):
    vertical_up = 0
    vertical_down = 0
    horizontal = 0

    for word in words:
        x0, y0, x1, y1 = word[:4]
        dx = x1 - x0
        dy = y1 - y0

        if abs(dy) > abs(dx):  # Vertical orientation
            if dy > 0:
                vertical_down += 1  # Top to bottom â†’ rotated 90Â°
            else:
                vertical_up += 1    # Bottom to top â†’ rotated -90Â°
        else:
            horizontal += 1

    if vertical_down > vertical_up and vertical_down > horizontal:
        return -90  # Rotate counter-clockwise to fix 90Â° rotation
    elif vertical_up > vertical_down and vertical_up > horizontal:
        return 90   # Rotate clockwise to fix -90Â° rotation
    else:
        return 0    # No rotation needed
